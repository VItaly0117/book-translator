\texttt{\textbackslash{}mathcal\{F\}{[}/*g{]}\ =\ \textbackslash{}mathcal\{F\}{[}/{]}\textbackslash{}mathcal\{F\}{[}g{]}.}

(12.3)

\texttt{(f*g)(x)\ =\ \textbackslash{}frac\{1\}\{\textbackslash{}sqrt\{2\textbackslash{}pi\}\}\ \textbackslash{}int\_\{-\textbackslash{}infty\}\^{}\{+\textbackslash{}infty\}\ f(x-\textbackslash{}xi)\ g(\textbackslash{}xi)\ d\textbackslash{}xi.}
